<userInfo>        := <userName> <userAvatar>
<userName>        := 'userName:' <string>
<userAvatar>      := 'userAvatar:' <int>

<userData>        := <userInfo> <userIP> <userID> <userRooms>
<userIP>          := 'userIP:' <string>
<userID>          := 'userID:' <int>
<userRooms>       := <roomName> | <roomName> <userRooms>
<roomName>        := 'roomName:' <string>

<roomInfo>        := <roomName> ( <chatRoomInfo> | <gameRoomInfo> )
<chatRoomInfo>    := 'type:' 0
<gameRoomInfo>    := 'type:' 1 <gameInfo>
<roomName>	      := 'roomName:' <string>

<rooms>           := <room> | <room> <rooms>
<room>            := <roomName> <users>
<users>           := <userName> <userData> <users> | ""

<action>          := <emoteAction> <userData> | <gameAction> <userData>
<gameAction>      := <gameStateAction> | <gameplayAction>

<gameStateAction> := 'action:' ( 0 | 1 | 2 | 3 | 4 )
<gameplayAction>  := <rpsAction> | <connect4Action>

<rpsAction>       := 'action' : ( 0 | 1 | 2 | 3 )
<connect4Action>  := 'action' : ( 0 <tokenPlacement> | 1 <cardActivation> )
<tokenPlacement>  := 'row' : <int>
<cardActivation>  := 'card' : <int>

<gameInfo>        := <rpsInfo> | <connect4Info>
<rpsInfo>         := 'gameType:' 0
<connect4Info>    := 'gameType:' 1 <rowCount> <colCount>
<rowCount>        := 'rowCount:' <int>
<colCount>        := 'columnCount:' <int>

<gameData>        := <gameAttributes> <rpsData> |  <gameAttributes> <connect4Data>
<gameAttributes>  := <players> <gameState>
<players>         := <player> <players> | <player>
<player>          := <userName> <playerState>
<playerState>     := 'playerState' : ( 0 | 1 | 2 | 3 | 4 | 5 )
<gameState>       := 'gameState' : ( 0 | 1 | 2 )
<rpsData>         := <player> <rpsAction> | <player> <rpsAction> <rpsData>
